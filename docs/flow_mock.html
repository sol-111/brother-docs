レスポンス

Request（× 3回並列送信） { "sourceImageId": "si_001", // Bubble DB.デザインテンプレート画像 "eventName": "春季例大祭", // Bubble DB.行事データ "period": "2026年4月20日〜22日", "details": "境内にて...", "program": "10:00 神事...", "notes": "雨天時は..." }
Response（即座に返る / 非同期） { "generatedImageId": "gi_001", // Generated Image DB.generatedImageId "sourceImageId": "si_001", "status": "PENDING", // Generated Image DB.status "fileKey": null, "thumbnailUrl": null }

Request（変更後の行事情報） // テンプレ・サイズは元の生成結果から引き継ぐ { "eventName": "春季例大祭", "period": "2026年4月20日〜4月23日", "details": "境内にて...", "program": "10:00 神事.../ 15:00 閉会", "notes": "雨天時は..." }
